\section*{Solution to Question 6}

\begin{enumerate}[label=(\alph*)]
    \item \textbf{Confidence Interval for $p$}
    Given: $n = 300, x = 210, \hat{p} = \displaystyle\frac{210}{300} = 0.70$.
    \begin{formula}{Confidence Interval for a Population Proportion}
        $\hat{p} \pm Z_{\alpha/2} \sqrt{\displaystyle\frac{\hat{p}(1-\hat{p})}{n}}$ \\
        $0.70 \pm 1.96 \sqrt{\displaystyle\frac{0.7 \times 0.3}{300}}$ \\
        $0.70 \pm 0.0519$ \\
        $\text{Interval:} \quad (0.6481, 0.7519)$
    \end{formula}

    \item \textbf{Interpretation}
    We are $95\%$ confident that the true proportion of all UCF students who support extending library hours until 2:00 AM is between \textbf{$64.81\%$ and $75.19\%$}.
\end{enumerate}
