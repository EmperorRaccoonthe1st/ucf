\section*{Solution to Question 1}

\begin{enumerate}[label=(\alph*)]
    \item \textbf{Identifying Components}
    \begin{enumerate}[label=\roman*.]
        \item \textbf{Population:} All UCF students.
        \item \textbf{Sample:} The $150$ UCF students who volunteered to participate in the initiative.
        \item \textbf{Parameter:} The true average weekly commute time of all UCF students ($\mu = 22$ minutes).
        \item \textbf{Statistic:} The average weekly commute time of the $150$ students after the initiative ($\bar{x} = 18$ minutes).
    \end{enumerate}

    \item \textbf{Classification of Variables}
    \begin{enumerate}[label=\roman*.]
        \item \textbf{Weekly commute time:} Quantitative, Continuous.
        \item \textbf{Primary mode of transportation:} Qualitative.
        \item \textbf{Number of days a student commutes:} Quantitative, Discrete.
        \item \textbf{Whether a student uses the initiative:} Qualitative.
    \end{enumerate}

    \item \textbf{Levels of Measurement}
    \begin{enumerate}[label=\roman*.]
        \item \textbf{Weekly commute time:} Ratio.
        \item \textbf{Primary mode of transportation:} Nominal.
        \item \textbf{Number of days a student commutes:} Ratio.
    \end{enumerate}

    \item \textbf{Study Type}
    \begin{itemize}
        \item \textbf{Answer:} \textbf{Observational Study}.
        \item \textbf{Justification:} The researcher did not randomly assign students to the carpooling initiative; instead, students volunteered (self-selection). The researcher is observing the outcomes of a group that chose to participate.
    \end{itemize}

    \item \textbf{Causation}
    \begin{itemize}
        \item \textbf{Answer:} No.
        \item \textbf{Explanation:} Because this is an observational study with volunteers, there may be confounding variables (e.g., highly motivated students). Without a control group and random assignment, the researcher cannot establish a cause-and-effect relationship.
    \end{itemize}

    \item \textbf{Sampling Method}
    \begin{enumerate}[label=\roman*.]
        \item \textbf{Method:} Simple Random Sampling.
        \item \textbf{Why preferred:} It reduces selection bias (like volunteer bias) and ensures the sample is more representative of the entire population, allowing for better generalization of results.
    \end{enumerate}
\end{enumerate}
