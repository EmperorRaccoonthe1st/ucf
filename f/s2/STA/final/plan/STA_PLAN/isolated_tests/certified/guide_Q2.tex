\section*{Study Topic 2: Descriptive Statistics and Visualization}

\begin{concept}{Descriptive Statistics}
    The \textbf{five-number summary} describes a dataset using: Minimum, $Q_1$, Median, $Q_3$, and Maximum. Data must be ordered from smallest to largest.
\end{concept}

\begin{concept}{Data Visualization (Boxplot)}
    A boxplot visually displays the five-number summary. The \textbf{Interquartile Range (IQR)} is used to identify outliers.
    \begin{formula}{IQR and Fences}
        $\displaystyle IQR = Q_3 - Q_1$ \\[0.5em]
        $\displaystyle \text{Lower Fence} = Q_1 - 1.5(IQR)$ \\[0.5em]
        $\displaystyle \text{Upper Fence} = Q_3 + 1.5(IQR)$
    \end{formula}
\end{concept}

\begin{concept}{Shape of Distribution}
    If the median is closer to $Q_1$, the distribution is \textbf{right-skewed}. If closer to $Q_3$, it is \textbf{left-skewed}.
\end{concept}
