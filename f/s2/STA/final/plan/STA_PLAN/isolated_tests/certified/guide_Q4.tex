\section*{Study Topic 4: Random Variables and Distributions}

\begin{concept}{Discrete Random Variables}
    A probability distribution lists all outcomes $x$ and their probabilities $P(x)$.
    Requirements: $0 \le P(x) \le 1$ and $\sum P(x) = 1$.
\end{concept}

\begin{concept}{Expected Value}
    The mean ($\mu$) of a discrete random variable, representing the long-term average.
    \begin{formula}{Expected Value}
        $E(X) = \sum [x \times P(x)]$
    \end{formula}
\end{concept}

\begin{concept}{Variance and Standard Deviation}
    Measures the average spread of outcomes around the expected value.
    \begin{formula}{Discrete Standard Deviation}
        $\sigma = \sqrt{\sum [x^2 \times P(x)] - \mu^2}$
    \end{formula}
\end{concept}

\begin{concept}{Binomial Distribution}
    Fixed trials $n$, independent trials, two outcomes, constant probability $p$.
    \begin{formula}{Binomial Probability}
        $\displaystyle P(X=x) = \binom{n}{x} p^x (1-p)^{n-x}$
    \end{formula}
\end{concept}

\begin{concept}{Binomial Properties}
    \begin{formula}{Binomial Mean and SD}
        $\mu = n \times p$ \\[0.5em]
        $\sigma = \sqrt{n \times p \times (1-p)}$
    \end{formula}
\end{concept}
