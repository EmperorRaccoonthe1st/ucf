\section*{Study Topic 5: Sampling Distributions}

\begin{concept}{Sampling Distribution of the Mean}
    Describes the distribution of all possible sample means.
    \begin{formula}{Mean and Standard Error}
        $\displaystyle \mu_{\bar{x}} = \mu$ \\[0.5em]
        $\displaystyle \sigma_{\bar{x}} = \frac{\sigma}{\sqrt{n}}$
    \end{formula}
\end{concept}

\begin{concept}{Central Limit Theorem}
    The distribution of $\bar{x}$ is approximately normal if $n \ge 30$ or the population is normal.
    \begin{formula}{Z-score for Means}
        $\displaystyle Z = \frac{\bar{x} - \mu_{\bar{x}}}{\sigma_{\bar{x}}}$
    \end{formula}
\end{concept}

\begin{concept}{Sampling Distribution of Proportion (Conditions)}
    Normal approximation for $\hat{p}$ requires:
    \begin{formula}{Success-Failure Condition}
        $n \times p \ge 10 \text{ and } n \times (1-p) \ge 10$
    \end{formula}
\end{concept}

\begin{concept}{Sampling Distribution of Proportion (Parameters)}
    \begin{formula}{Mean and Standard Error for Proportions}
        $\displaystyle \mu_{\hat{p}} = p$ \\[0.5em]
        $\displaystyle \sigma_{\hat{p}} = \sqrt{\frac{p(1-p)}{n}}$
    \end{formula}
\end{concept}

\begin{concept}{Sampling Distribution of Proportion (Probability)}
    Calculating the probability of a sample proportion.
    \begin{formula}{Z-score for Proportions}
        $\displaystyle Z = \frac{\hat{p} - \mu_{\hat{p}}}{\sigma_{\hat{p}}}$
    \end{formula}
\end{concept}
