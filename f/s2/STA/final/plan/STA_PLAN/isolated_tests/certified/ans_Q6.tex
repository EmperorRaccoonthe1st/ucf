\section*{Question 6}

\begin{enumerate}[label=(\alph*)]
    \item \begin{formula}{Confidence Interval for Proportion}
        $\displaystyle \hat{p} \pm z_{\alpha/2} \sqrt{\frac{\hat{p}(1-\hat{p})}{n}}$
    \end{formula}
    Given: $n = 250, x = 165 \Rightarrow \hat{p} = 0.66, z_{0.025} = 1.96$ \par\vspace{0.5em}
    $\displaystyle \text{Margin of Error (E)} = 1.96 \sqrt{\frac{0.66(0.34)}{250}} \approx 1.96(0.02996) \approx 0.0587$ \par\vspace{0.5em}
    CI: $0.66 \pm 0.0587 = (0.6013, 0.7187)$

    \item \textbf{Interpretation:}
    We are 95\% confident that the true proportion of all students who support having flu vaccination clinics on campus every semester lies between 60.13\% and 71.87\%.
\end{enumerate}
