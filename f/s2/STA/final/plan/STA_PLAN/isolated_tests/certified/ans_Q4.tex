\section*{Question 4}

\begin{enumerate}[label=(\alph*)]
    \item \textbf{Probability Distribution of $X$:}
    \begin{itemize}
        \item $x = -2$, $\displaystyle P(X = -2) = \frac{2}{6} = 0.3333$
        \item $x = 1$, $\displaystyle P(X = 1) = \frac{3}{6} = 0.5000$
        \item $x = 5$, $\displaystyle P(X = 5) = \frac{1}{6} = 0.1667$
    \end{itemize}

    \item \begin{formula}{Expected Value}
        $E(X) = \sum [x \times P(X=x)]$
    \end{formula}
    $\displaystyle E(X) = (-2)(0.3333) + (1)(0.5000) + (5)(0.1667) \approx 0.6667$

    \item \begin{formula}{Standard Deviation of a Discrete Random Variable}
        $\sigma = \sqrt{\sum [x^2 \times P(x)] - [E(X)]^2}$
    \end{formula}
    $E(X^2) = (-2)^2(1/3) + (1)^2(1/2) + (5)^2(1/6) = 4/3 + 1/2 + 25/6 = 8/6 + 3/6 + 25/6 = 36/6 = 6$ \\[0.5em]
    $\sigma = \sqrt{6 - (2/3)^2} = \sqrt{6 - 4/9} = \sqrt{50/9} \approx 2.3570$

    \item \begin{formula}{Binomial Probability}
        $\displaystyle P(Y=y) = \binom{n}{y} p^y (1-p)^{n-y}$
    \end{formula}
    $n = 10, p = 0.5, P(Y \ge 3) = 1 - P(Y \le 2)$ \\[0.5em]
    $P(Y \le 2) = P(Y=0) + P(Y=1) + P(Y=2) = 0.0009765 + 0.0097656 + 0.0439453 = 0.0546875$ \\[0.5em]
    $P(Y \ge 3) = 1 - 0.0546875 \approx 0.9453$

    \item \begin{formula}{Binomial Mean and Standard Deviation}
        $\mu_Y = n \times p$ \\
        $\sigma_Y = \sqrt{n \times p \times (1-p)}$
    \end{formula}
    $\mu_Y = 10 \times 0.5 = 5$ \\[0.5em]
    $\sigma_Y = \sqrt{10 \times 0.5 \times 0.5} = \sqrt{2.5} \approx 1.5811$
\end{enumerate}
